		array = KOR_legco_faction_support
		index = i
		value = val

		if = {
			limit = { check_variable = { i < 14 } } #정당 수 + 1

		var = KOR_legco_faction_support^KOR_legco_faction
		value = KOR_temp_votes
	}
	set_temp_variable = { KOR_temp_votes_color = KOR_temp_votes }

		limit = {
			check_variable = { KOR_temp_votes < 0 }
